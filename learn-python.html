# PARENT CLASS
class Animal:
    kingdom = "Animalia"   # Class attribute (shared)

    def __init__(self, name, age):
        self.name = name   # Instance attribute (unique)
        self.age = age
        self._energy = 100 # "Protected" by convention

    def speak(self):
        return "..."

    def __repr__(self):    # How it looks in print()
        return f"Animal({self.name}, {self.age})"


# CHILD CLASS (inherits Animal)
class Dog(Animal):
    def __init__(self, name, age, breed):
        super().__init__(name, age)   # Call parent init
        self.breed = breed

    def speak(self):               # Override parent method
        return f"{self.name} barks: Woof!"

    @property                         # Computed property
    def info(self):
        return f"{self.name} ({self.breed}, {self.age} yrs)"


# USAGE
dog = Dog("Buddy", 3, "Labrador")
print(dog.speak())
print(dog.info)                   # Property (no parentheses)
print(f"Kingdom: {dog.kingdom}") # Inherited class attribute

# Polymorphism — same interface, different behaviour
class Cat(Animal):
    def speak(self): return f"{self.name} says: Meow!"

for animal in [Dog("Rex", 2, "Poodle"), Cat("Luna", 4)]:
    print(animal.speak())

# Full try-except-else-finally structure
try:
    result = 10 / 2
    data = {"key": "value"}
    print("Operation succeeded")

except ZeroDivisionError:
    print("Can't divide by zero!")

except (KeyError, TypeError) as e:
    print(f"Key or type issue: {e}")

except Exception as e:      # Catch-all (use sparingly)
    print(f"Unexpected: {e}")

else:                         # Runs ONLY if no exception
    print(f"Result: {result}")

finally:                      # ALWAYS runs (cleanup)
    print("--- Done (finally) ---")


# Custom exceptions
class InsufficientFundsError(Exception):
    def __init__(self, amount, balance):
        super().__init__(f"Need {amount} but have {balance}")

def withdraw(balance, amount):
    if amount > balance:
        raise InsufficientFundsError(amount, balance)
    return balance - amount

try:
    withdraw(100, 500)
except InsufficientFundsError as e:
    print(f"Bank Error: {e}")

# ── LIST COMPREHENSION ─────────────────
squares  = [x**2 for x in range(10)]
evens    = [x    for x in range(20) if x % 2 == 0]
upper_w  = [w.upper() for w in ["hello", "world"]]

print("Squares:", squares)
print("Evens:",   evens)
print("Upper:",   upper_w)

# ── DICT COMPREHENSION ─────────────────
words    = ["python", "is", "awesome"]
word_len = {w: len(w) for w in words}
inverted = {v: k for k, v in word_len.items()}  # Flip!

print("\nWord lengths:", word_len)
print("Inverted:",     inverted)

# ── SET COMPREHENSION ──────────────────
unique_lens = {len(w) for w in words}
print("\nUnique lengths:", unique_lens)

# ── GENERATOR EXPRESSION ──────────────
# Like list comp but lazy — memory efficient for big data
gen = (print(x) or x**2 for x in range(5))
total = sum(x**2 for x in range(10))       # No list needed!
print(f"\nSum of squares 0-9: {total}")

# ── NESTED COMPREHENSION ──────────────
matrix = [[i*j for i in range(1, 4)] for j in range(1, 4)]
print("\nMultiplication matrix:")
for row in matrix:
    print(" ", row)
